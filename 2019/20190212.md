Guston syntax sketch

```
(= null?
  (fn (xs)
    (== xs (quote ()))))

(= lat?
  (fn (xs)
    (if (null? xs)
      (quote t)
      (if (atom? (first xs))
        (lat? (rest xs))
        (quote ())))))

(lat? (quote (a b c)))
; t

(= member?
  (fn (x xs)
    (if (null? xs)
      (quote t)
      (if (== x (first xs))
        (quote t)
        (member? x (rest xs))))))

(member? (quote c) (quote (a b c)))
; t

(= rember
  (fn (x xs)
    (if (null? xs)
      xs
      (if (== x (first xs))
        (rest xs)
        (::
          (first xs)
          (rember x (rest xs)))))))

(rember (quote b) (quote (a b c)))
; (a c)
```
